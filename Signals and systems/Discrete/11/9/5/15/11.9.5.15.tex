\textbf{Question 11.9.5.15:}
The pth, qth and rth terms of an AP are a,b,c respectively. Show that
\begin{align*} \brak{q-r}a + \brak{r-p}b +\brak{p-q}c =0 \end{align*}
\solution\\
The AP has the following parameters
\begin{table}[ht!]
    \centering
    \input{tables/table1.tex}
    \caption{Input Parameters}
\end{table}

Now,
\begin{align}
    x\brak{p-1} &= x\brak{0}+\brak{p-1}d = a \label{eq:1} \\
    x\brak{q-1} &= x\brak{0}+\brak{q-1}d = b \label{eq:2} \\
    x\brak{r-1} &= x\brak{0}+\brak{r-1}d = c \label{eq:3} \\
    p-q &= \frac{a-b}{d} \label{eq:4} \\
    q-r &= \frac{b-c}{d} \label{eq:5} \\
    r-p &= \frac{c-a}{d} \label{eq:6}
\end{align}
Now, from \eqref{eq:4}, \eqref{eq:5} and \eqref{eq:6},
\begin{align}
    \brak{q-r}a + \brak{r-p}b + \brak{p-q}c = \frac{b-c}{d}.a +  \frac{c-a}{d}.b + \frac{a-b}{d}.c \label{eq:7}
\end{align}
\begin{align*}
    &= 0
\end{align*}
Hence proved
